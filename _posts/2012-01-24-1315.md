<pre><code>  appenders {

      syslogHost: "127.0.0.1", // &lt;&lt; Host
      facility: org.apache.log4j.net.SyslogAppender.LOG&amp;#95;LOCAL3 // &lt;&lt; Logging queue
    )
  }
</code></pre>
